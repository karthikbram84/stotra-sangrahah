\sect{गोदाष्टोत्तरशतनामस्तोत्रम्}


\dnsub{ध्यानम्}
\fourlineindentedshloka*
{शतमखमणि  नीला  चारुकल्हारहस्ता}
{स्तनभरनमिताङ्गी  सान्द्रवात्सल्यसिन्धुः}
{अलकविनिहिताभिः  स्रग्भिराकृष्टनाथा}
{विलसतु  हृदि  गोदा  विष्णुचित्तात्मजा  नः}

\dnsub{स्तोत्रम्}
\twolineshloka
{श्रीरङ्गनायकी  गोदा विष्णुचित्तात्मजा सती}
{गोपीवेषधरा  देवी  भूसुता  भोगशालिनी}%1

\twolineshloka
{तुलसीकाननोद्भूता  श्रीधन्विपुरवासिनी}
{भट्टनाथप्रियकरी  श्रीकृष्णहितभोगिनी}%2

\twolineshloka
{आमुक्तमाल्यदा  बाला  रङ्गनाथप्रिया  परा}
{विश्वम्भरा  कलालापा  यतिराजसहोदरी}%3

\twolineshloka
{कृष्णानुरक्ता  सुभगा  सुलभश्रीः  सलक्षणा}
{लक्ष्मीप्रियसखी  श्यामा  दयाञ्चितदृगञ्चला}%4

\twolineshloka
{फल्गुन्याविर्भवा  रम्या  धनुर्मासकृतव्रता}
{चम्पकाशोक-पुन्नाग-मालती-विलसत्-कचा}%5

\twolineshloka
{आकारत्रयसम्पन्ना  नारायणपदाश्रिता}
{श्रीमदष्टाक्षरीमन्त्र-राजस्थित-मनोरथा}%6

\twolineshloka
{मोक्षप्रदाननिपुणा  मनुरत्नाधिदेवता}
{ब्रह्मण्या  लोकजननी  लीलामानुषरूपिणी}%7

\twolineshloka
{ब्रह्मज्ञानप्रदा माया  सच्चिदानन्दविग्रहा}
{महापतिव्रता  विष्णुगुणकीर्तनलोलुपा}%8

\twolineshloka
{प्रपन्नार्तिहरा  नित्या  वेदसौधविहारिणी}
{श्रीरङ्गनाथमाणिक्यमञ्जरी  मञ्जुभाषिणी}%9

\twolineshloka
{पद्मप्रिया पद्महस्ता वेदान्तद्वयबोधिनी}
{सुप्रसन्ना भगवती श्रीजनार्दनदीपिका}%10

\twolineshloka
{सुगन्धवयवा चारुरङ्गमङ्गलदीपिका}
{ध्वजवज्राङ्कुशाब्जाङ्क-मृदुपाद-लताञ्चिता}%11

\twolineshloka
{तारकाकारनखरा  प्रवालमृदुलाङ्गुली}
{कूर्मोपमेय-पादोर्ध्वभागा  शोभनपार्ष्णिका}%12

\twolineshloka
{वेदार्थभावतत्त्वज्ञा लोकाराध्याङ्घ्रिपङ्कजा}
{आनन्दबुद्बुदाकार-सुगुल्फा  परमाऽणुका}%13

\twolineshloka
{तेजःश्रियोज्ज्वलधृतपादाङ्गुलि-सुभूषिता}
{मीनकेतन-तूणीर-चारुजङ्घा-विराजिता}%14

\twolineshloka
{ककुद्वज्जानुयुग्माढ्या  स्वर्णरम्भाभसक्थिका}
{विशालजघना  पीनसुश्रोणी  मणिमेखला}%15

\twolineshloka
{आनन्दसागरावर्त-गम्भीराम्भोज-नाभिका}
{भास्वद्बलित्रिका  चारुजगत्पूर्ण-महोदरी}%16

\twolineshloka
{नववल्लीरोमराजी  सुधाकुम्भायितस्तनी}
{कल्पमालानिभभुजा  चन्द्रखण्ड-नखाञ्चिता}%17

\twolineshloka
{सुप्रवाशाङ्गुलीन्यस्तमहारत्नाङ्गुलीयका}
{नवारुणप्रवालाभ-पाणिदेश-समञ्चिता}%18

\twolineshloka
{कम्बुकण्ठी  सुचुबुका  बिम्बोष्ठी  कुन्ददन्तयुक्}
{कारुण्यरस-निष्यन्द-नेत्रद्वय-सुशोभिता}%19


\twolineshloka
{मुक्ताशुचिस्मिता  चारुचाम्पेयनिभनासिका}
{दर्पणाकार-विपुल-कपोल-द्वितयाञ्चिता}%20


\twolineshloka
{अनन्तार्क-प्रकाशोद्यन्मणि-ताटङ्क-शोभिता}
{कोटिसूर्याग्निसङ्काश-नानाभूषण-भूषिता}%21


\twolineshloka
{सुगन्धवदना  सुभ्रू  अर्धचन्द्रललाटिका}
{पूर्णचन्द्रानना  नीलकुटिलालकशोभिता}%22


\twolineshloka
{सौन्दर्यसीमा  विलसत्-कस्तूरी-तिलकोज्ज्वला}
{धगद्ध-गायमानोद्यन्मणि-सीमन्त-भूषणा}%23


\twolineshloka
{जाज्वल्यमान-सद्रत्न-दिव्यचूडावतंसका}
{सूर्यार्धचन्द्र-विलसत्-भूषणाञ्चित-वेणिका}%24


\twolineshloka
{अत्यर्कानल-तेजोधिमणि-कञ्चुकधारिणी}
{सद्रत्नाञ्चितविद्योत-विद्युत्कुञ्जाभ-शाटिका}%25

\twolineshloka
{नानामणिगणाकीर्ण-हेमाङ्गदसुभूषिता}
{कुङ्कुमागरु-कस्तूरी-दिव्यचन्दन-चर्चिता}%26

\twolineshloka
{स्वोचितौज्ज्वल्य-विविध-विचित्र-मणि-हारिणी}
{असङ्ख्येय-सुखस्पर्श-सर्वातिशय-भूषणा}%27

\twolineshloka
{मल्लिका-पारिजातादि  दिव्यपुष्प-स्रगञ्चिता}
{श्रीरङ्गनिलया  पूज्या  दिव्यदेशसुशोभिता}%28

{॥इति श्री-गोदाष्टोत्तरशतनामस्तोत्रं सम्पूर्णम्॥}
